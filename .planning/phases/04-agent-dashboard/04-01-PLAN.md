# Dashboard Center Area + Left Panel

## Goal

实现 Dashboard 页面的核心功能：中心区域的 KPI 摘要条、搜索筛选器和 Agent 卡片网格，以及左侧面板的子导航标签（Overview/Agents/Skills）。用户可以实时查看所有 Agent 状态，通过搜索/筛选快速定位目标 Agent，并通过左侧标签切换不同维度的数据视图。

## Files to Create

- `components/dashboard/dashboard-kpi-bar.tsx` — KPI 摘要条组件，显示活跃/工作中/错误 Agent 数和 Token 用量
- `components/dashboard/agent-search-filter.tsx` — 搜索框和状态筛选器组件
- `components/dashboard/agent-card.tsx` — 单个 Agent 卡片组件，使用 HudCard
- `components/dashboard/agent-card-grid.tsx` — Agent 卡片网格容器，响应式布局
- `components/dashboard/dashboard-left-panel.tsx` — 左侧面板容器，包含标签导航
- `components/dashboard/tab-nav.tsx` — 可复用的标签导航组件
- `components/dashboard/overview-tab.tsx` — Overview 标签内容，显示全局 stats 和 alert 列表
- `components/dashboard/agents-tab.tsx` — Agents 标签内容，显示 Agent 列表
- `components/dashboard/skills-tab.tsx` — Skills 标签内容，显示技能列表
- `components/dashboard/empty-state.tsx` — 可复用的空状态组件

## Files to Modify

- `app/(shell)/dashboard/page.tsx` — 替换占位内容，实现完整的 Dashboard 页面布局
- `app/globals.css` — 添加必要的自定义 CSS（如果需要）

```tsx
'use client'

import { useGatewayStore } from '@/stores/gateway/gateway-store'
import { HudCard } from '@/components/hud/hud-card'
import { cn } from '@/lib/utils'

interface KpiCardProps {
  label: string
  value: string | number
  className?: string
}

function KpiCard({ label, value, className }: KpiCardProps) {
  return (
    <HudCard variant="sm" className={cn('flex items-center gap-2 px-3 py-2', className)}>
      <span className="text-xs text-muted-foreground">{label}</span>
      <span className="text-sm font-semibold text-foreground">{value}</span>
    </HudCard>
  )
}

export function DashboardKpiBar() {
  const agents = useGatewayStore((state) => state.agents)
  const providers = useGatewayStore((state) => state.providers)
  const activeSessions = useGatewayStore((state) => state.activeSessions)

  // Calculate stats
  const agentList = Array.from(agents.values())
  const workingCount = agentList.filter((a) => a.status === 'working' || a.status === 'tool_calling').length
  const errorCount = agentList.filter((a) => a.status === 'error').length
  const totalTokens = providers.reduce((sum, p) => sum + (p.totalTokens || 0), 0)

  return (
    <div className="flex items-center gap-3 p-4 border-b border-border">
      <KpiCard label="Active" value={agentList.length} />
      <KpiCard label="Working" value={workingCount} />
      <KpiCard label="Errors" value={errorCount} className="text-destructive" />
      <KpiCard label="Tokens" value={totalTokens.toLocaleString()} />
      <KpiCard label="Sessions" value={activeSessions} />
    </div>
  )
}
```

### Step 4: Create AgentSearchFilter component

**File:** `components/dashboard/agent-search-filter.tsx`

**What:** 搜索框和状态筛选器，支持按名称搜索（300ms debounce）和按状态筛选

**Detail:**
```tsx
'use client'

import { useState, useCallback } from 'react'
import { Input } from '@/components/ui/input'
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from '@/components/ui/select'
import { Search } from 'lucide-react'
import { cn } from '@/lib/utils'
import type { AgentDisplayStatus } from '@/stores/gateway/gateway-store'

interface AgentSearchFilterProps {
  searchQuery: string
  onSearchChange: (query: string) => void
  statusFilter: AgentDisplayStatus | 'all'
  onStatusChange: (status: AgentDisplayStatus | 'all') => void
  className?: string
}

export function AgentSearchFilter({
  searchQuery,
  onSearchChange,
  statusFilter,
  onStatusChange,
  className,
}: AgentSearchFilterProps) {
  const [localSearch, setLocalSearch] = useState(searchQuery)

  // 300ms debounce for search
  const debouncedSearch = useCallback(
    (value: string) => {
      const timer = setTimeout(() => {
        onSearchChange(value)
      }, 300)
      return () => clearTimeout(timer)
    },
    [onSearchChange]
  )

  const handleSearchChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    const value = e.target.value
    setLocalSearch(value)
    debouncedSearch(value)
  }

  return (
    <div className={cn('flex items-center gap-3 p-4 border-b border-border', className)}>
      <div className="relative flex-1">
        <Search className="absolute left-3 top-1/2 -translate-y-1/2 h-4 w-4 text-muted-foreground" />
        <Input
          type="text"
          placeholder="Search agents..."
          value={localSearch}
          onChange={handleSearchChange}
          className="pl-9 h-8 bg-card border-border"
        />
      </div>
      <Select value={statusFilter} onValueChange={onStatusChange}>
        <SelectTrigger className="w-[140px] h-8 bg-card border-border">
          <SelectValue placeholder="Filter by status" />
        </SelectTrigger>
        <SelectContent>
          <SelectItem value="all">All Status</SelectItem>
          <SelectItem value="idle">Idle</SelectItem>
          <SelectItem value="working">Working</SelectItem>
          <SelectItem value="tool_calling">Tool Calling</SelectItem>
          <SelectItem value="speaking">Speaking</SelectItem>
          <SelectItem value="error">Error</SelectItem>
        </SelectContent>
      </Select>
    </div>
  )
}
```

```tsx
import { HudCard } from '@/components/hud/hud-card'
import { Badge } from '@/components/ui/badge'
import { cn } from '@/lib/utils'
import type { AgentInfo } from '@/stores/gateway/gateway-store'

interface AgentCardProps {
  agent: AgentInfo
  isSelected?: boolean
  onClick?: () => void
  className?: string
}

// Agent status colors (OKLCH tokens for consistency)
const statusColors: Record<AgentInfo['status'], string> = {
  idle: 'oklch(0.55 0.008 160)',        // gray-500
  working: 'oklch(0.62 0.17 65)',       // blue-500 (accent)
  tool_calling: 'oklch(0.8 0.17 75)',   // yellow-500 (accent)
  speaking: 'oklch(0.65 0.15 145)',     // green-500
  error: 'oklch(0.577 0.245 27.325)',   // red-500 (destructive)
}

export function AgentCard({ agent, isSelected = false, onClick, className }: AgentCardProps) {
  const statusColor = statusColors[agent.status]

  return (
    <HudCard
      variant="sm"
      glow={isSelected}
      className={cn(
        'cursor-pointer transition-all hover:opacity-80',
        isSelected && 'ring-2 ring-accent',
        className
      )}
      onClick={onClick}
    >
      <div className="flex flex-col gap-2 p-3">
        {/* Avatar + Name */}
        <div className="flex items-center gap-2">
          {agent.avatarUrl ? (
            <img
              src={agent.avatarUrl}
              alt={agent.name}
              className="w-8 h-8 rounded bg-muted"
            />
          ) : (
            <div className="w-8 h-8 rounded bg-muted flex items-center justify-center text-xs font-semibold">
              {agent.name.charAt(0).toUpperCase()}
            </div>
          )}
          <div className="flex-1 min-w-0">
            <div className="text-sm font-semibold text-foreground truncate">{agent.name}</div>
          </div>
        </div>

        {/* Status indicator */}
        <div className="flex items-center gap-2">
          <div
            className="w-2 h-2 rounded-full"
            style={{ backgroundColor: statusColor }}
          />
          <span className="text-xs text-muted-foreground capitalize">{agent.status}</span>
        </div>

        {/* Current tool (if any) */}
        {agent.currentTool && (
          <Badge variant="secondary" className="text-xs">
            {agent.currentTool}
          </Badge>
        )}
      </div>
    </HudCard>
  )
}
```

### Step 6: Create AgentCardGrid component

**File:** `components/dashboard/agent-card-grid.tsx`

**What:** Agent 卡片网格容器，响应式布局（auto-fill, minmax 180px）

**Detail:**
```tsx
'use client'

import { useMemo } from 'react'
import { AgentCard } from './agent-card'
import { EmptyState } from './empty-state'
import { cn } from '@/lib/utils'
import type { AgentInfo } from '@/stores/gateway/gateway-store'

interface AgentCardGridProps {
  agents: AgentInfo[]
  selectedAgentId: string | null
  onAgentClick: (agentId: string) => void
  searchQuery: string
  statusFilter: AgentInfo['status'] | 'all'
  className?: string
}

export function AgentCardGrid({
  agents,
  selectedAgentId,
  onAgentClick,
  searchQuery,
  statusFilter,
  className,
}: AgentCardGridProps) {
  // Filter and sort agents
  const filteredAgents = useMemo(() => {
    let filtered = agents

    // Filter by status
    if (statusFilter !== 'all') {
      filtered = filtered.filter((a) => a.status === statusFilter)
    }

    // Filter by search query
    if (searchQuery.trim()) {
      const query = searchQuery.toLowerCase()
      filtered = filtered.filter((a) =>
        a.name.toLowerCase().includes(query)
      )
    }

    // Sort: active first, then by name
    return filtered.sort((a, b) => {
      const aActive = a.status !== 'idle'
      const bActive = b.status !== 'idle'
      if (aActive && !bActive) return -1
      if (!aActive && bActive) return 1
      return a.name.localeCompare(b.name)
    })
  }, [agents, searchQuery, statusFilter])

  if (filteredAgents.length === 0) {
    return (
      <EmptyState
        title={searchQuery || statusFilter !== 'all' ? 'No matching agents' : 'No agents found'}
        description={
          searchQuery || statusFilter !== 'all'
            ? 'Try adjusting your search or filter'
            : 'Connect to Gateway to see agents'
        }
        className={cn('h-full min-h-[200px]', className)}
      />
    )
  }

  return (
    <div
      className={cn(
        'grid gap-3 p-4',
        'grid-cols-[repeat(auto-fill,minmax(180px,1fr))]',
        className
      )}
    >
      {filteredAgents.map((agent) => (
        <AgentCard
          key={agent.id}
          agent={agent}
          isSelected={selectedAgentId === agent.id}
          onClick={() => onAgentClick(agent.id)}
        />
      ))}
    </div>
  )
}
```

## Store Integration

本计划消费以下 store 状态和 selectors：

**Raw state (from `useGatewayStore()`):**
- `agents: Map<string, AgentInfo>` — Agent 列表数据
- `providers: UsageProviderInfo[]` — Token 用量数据
- `activeSessions: number` — 活跃会话数
- `skills: SkillInfo[]` — 技能列表
- `connectionStatus: ConnectionStatus` — Gateway 连接状态

**Selectors (from `p0-selectors.ts`):**
- `selectAlertsState(state)` — 返回 `{ state: P0UIState, data: AlertItem[] }`，用于 OverviewTab 的 alert 列表

**不使用 selector 的场景:**
- Agent 列表直接从 `agents` Map 转换为数组（无需过滤/状态判断）
- KPI 统计直接从原始数据计算（无需额外的状态机）
