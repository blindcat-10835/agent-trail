    ```typescript
    import { NextRequest } from 'next/server';
    import { assertSourceToolId } from '@/lib/agent-tools/registry';

    export async function GET(
      request: NextRequest,
      { params }: { params: Promise<{ tool: string }> },
    ) {
      const { tool } = await params;
      try {
        assertSourceToolId(tool);

        // Proxy SSE stream from ingest
        const ingestUrl = process.env.INGEST_URL || 'http://localhost:8078';
        const ingestRes = await fetch(`${ingestUrl}/api/v1/events`, {
          headers: { Accept: 'text/event-stream' },
          // No cache — SSE is always live
          cache: 'no-store',
          // Pass through the abort signal so client disconnect propagates to ingest
          signal: request.signal,
        });

        if (!ingestRes.ok) {
          return new Response(JSON.stringify({ error: 'Ingest SSE unavailable' }), {
            status: 502,
            headers: { 'Content-Type': 'application/json' },
          });
        }

        // Stream the ingest SSE response directly to the client
        return new Response(ingestRes.body, {
          status: 200,
          headers: {
            'Content-Type': 'text/event-stream',
            'Cache-Control': 'no-cache',
            'Connection': 'keep-alive',
            'X-Accel-Buffering': 'no',
          },
        });
      } catch (err) {
        return new Response(JSON.stringify({ error: 'SSE proxy error' }), {
          status: 502,
          headers: { 'Content-Type': 'application/json' },
        });
      }
    }
    ```

    ```typescript
    import { NextRequest, NextResponse } from 'next/server';
    import { assertSourceToolId } from '@/lib/agent-tools/registry';
    import { openclawAdapter } from '@/lib/agent-tools/openclaw/server-adapter';
    import { sanitizeError } from '@/lib/agent-tools/server-adapter';

    export async function GET(
      request: NextRequest,
      { params }: { params: Promise<{ tool: string }> },
    ) {
      const { tool } = await params;
      try {
        const toolId = assertSourceToolId(tool);
        const key = request.nextUrl.searchParams.get('key');

        if (!key) {
          return NextResponse.json({ error: 'key parameter is required' }, { status: 400 });
        }

        // Only OpenClaw supports Gateway-based lookup (per CONTEXT.md)
        if (toolId !== 'openclaw') {
          return NextResponse.json(
            { error: 'Gateway lookup is only available for OpenClaw' },
            { status: 400 },
          );
        }

        const session = await openclawAdapter.lookupSessionByKey(key);
        if (!session) {
          return NextResponse.json(
            { error: 'No matching indexed session found', key },
            { status: 404 },
          );
        }
        return NextResponse.json(session);
      } catch (err) {
        const { error, code } = sanitizeError(err);
        return NextResponse.json({ error }, { status: code });
      }
    }
    ```

    **1. `useSSE(toolId: AgentToolId, sessionId?: string, onEvent?: (event: SSEEvent) => void)`:**

    ```typescript
    'use client';

    interface SSEEvent {
      event: string;
      data: Record<string, unknown>;
    }

    export function useSSE(
      toolId: AgentToolId,
      sessionId?: string,
      onEvent?: (event: SSEEvent) => void,
    ) {
      const [connected, setConnected] = useState(false);
      const reconnectTimerRef = useRef<ReturnType<typeof setTimeout>>();

      useEffect(() => {
        let es: EventSource | null = null;
        let retries = 0;
        const maxRetries = 10;

        function connect() {
          const url = sessionId
            ? `/api/agent-tools/${toolId}/events?sessionId=${sessionId}`
            : `/api/agent-tools/${toolId}/events`;

          es = new EventSource(url);

          es.onopen = () => {
            setConnected(true);
            retries = 0;
          };

          es.onmessage = (msg) => {
            try {
              const event: SSEEvent = {
                event: msg.type || 'message',
                data: JSON.parse(msg.data),
              };
              onEvent?.(event);
            } catch { /* ignore parse errors */ }
          };

          // Handle named events via addEventListener
          for (const eventType of ['session_created', 'session_updated', 'session_removed', 'sync_complete', 'turn_added']) {
            es.addEventListener(eventType, (msg: MessageEvent) => {
              try {
                onEvent?.({ event: eventType, data: JSON.parse(msg.data) });
              } catch { /* ignore */ }
            });
          }

          es.onerror = () => {
            setConnected(false);
            es?.close();
            es = null;
            if (retries < maxRetries) {
              retries++;
              const backoff = 3000 * Math.min(retries, 5); // 3s, 6s, 9s, 12s, 15s max
              reconnectTimerRef.current = setTimeout(connect, backoff);
            }
          };
        }

        connect();

        return () => {
          if (reconnectTimerRef.current) clearTimeout(reconnectTimerRef.current);
          es?.close();
        };
      }, [toolId, sessionId]);

      return { connected };
    }
    ```

    ```typescript
    type IngestStatus = 'connected' | 'disconnected' | 'reconnecting' | 'loading';

    export function useIngestStatus(toolId: AgentToolId): IngestStatus {
      const [status, setStatus] = useState<IngestStatus>('loading');

      useEffect(() => {
        let attempts = 0;
        let timer: ReturnType<typeof setTimeout>;

        function check() {
          fetchToolApi<{ status: string; version?: string }>(toolId, '/health')
            .then((data) => {
              setStatus(data.status === 'ok' ? 'connected' : 'disconnected');
              attempts = 0;
              timer = setTimeout(check, 30000); // re-check every 30s
            })
            .catch(() => {
              attempts++;
              if (attempts <= 1) {
                setStatus('reconnecting');
              } else {
                setStatus('disconnected');
              }
              timer = setTimeout(check, 3000 * Math.min(attempts, 5));
            });
        }

        check();
        return () => clearTimeout(timer);
      }, [toolId]);

      return status;
    }
    ```
