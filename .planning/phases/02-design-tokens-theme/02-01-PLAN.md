    **完整文件内容：**
    ```typescript
    import { create } from 'zustand'
    import { persist } from 'zustand/middleware'

    type Theme = 'light' | 'dark' | 'system'

    interface ThemeStore {
      theme: Theme
      setTheme: (theme: Theme) => void
      resolvedTheme: 'light' | 'dark'
      getResolvedTheme: () => 'light' | 'dark'
    }

    export const useThemeStore = create<ThemeStore>()(
      persist(
        (set, get) => ({
          theme: 'system',
          setTheme: (theme) => {
            set({ theme })
            const resolved = theme === 'system' 
              ? (window.matchMedia('(prefers-color-scheme: dark)').matches ? 'dark' : 'light')
              : theme
            set({ resolvedTheme: resolved })
            document.documentElement.setAttribute('data-theme', resolved)
          },
          resolvedTheme: typeof window !== 'undefined' && window.matchMedia('(prefers-color-scheme: dark)').matches ? 'dark' : 'light',
          getResolvedTheme: () => {
            const { theme } = get()
            if (theme === 'system') {
              return window.matchMedia('(prefers-color-scheme: dark)').matches ? 'dark' : 'light'
            }
            return theme
          },
        }),
        {
          name: 'theme-storage',
          partialize: (state) => ({ theme: state.theme }),
        }
      )
    )
    ```

          aria-label={resolvedTheme === 'dark' ? '切换到亮色主题' : '切换到暗色主题'}

        >
          {resolvedTheme === 'dark' ? (
            <>
              <Sun className="w-3.5 h-3.5" />
              <span>LIGHT</span>
            </>
          ) : (
            <>

              <span>DARK</span>

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Tampering | localStorage theme value | mitigate | Validate theme enum values in setTheme function ('light' | 'dark' | 'system'), fallback to 'system' if invalid/corrupt |
| T-02-02 | Spoofing | Theme toggle button (no auth needed) | accept | Client-side only, no server state modification — not applicable |
| T-02-03 | Information Disclosure | localStorage pollution | mitigate | Validate theme value on read from localStorage in persist middleware, fallback to 'system' default |
| T-02-04 | Tampering | data-theme attribute injection | mitigate | Strict TypeScript typing prevents arbitrary values, setTheme only accepts 'light' | 'dark' | 'system' |

**Security note:** Theme system is client-side only with no API calls. Primary risk is XSS via malicious localStorage values. Mitigated by strict TypeScript typing + runtime enum validation.
