```typescript
export type AgentToolId = 'openclaw' | 'claude-code' | 'codex'

export interface AgentToolDefinition {
  id: AgentToolId
  label: string           // Full display name
  shortLabel: string       // Abbreviated label for tight spaces
  defaultRoute: string     // e.g., '/dashboard'
  capabilities: AgentToolCapabilities
  nav: ToolNavItem[]
  ui: AgentToolUIProfile
}

export interface AgentToolCapabilities {
  liveGateway: boolean    // OpenClaw only
  sessions: boolean
  replay: boolean
  activity: boolean
  office: boolean
  workspace: boolean
  subagents: boolean
  cost: boolean
  approvals: boolean
}

export interface ToolNavItem {
  id: string
  href: (toolId: AgentToolId) => string
  label: string            // Short abbv for sidebar (e.g., 'OVR', 'SES')
  title: string             // Full tooltip label (e.g., 'Overview', 'Sessions')
  requiredCapability?: keyof AgentToolCapabilities
}

export interface AgentToolUIProfile {
  brand: {
    name: string
    versionLabel?: string
    accentToken?: string
  }
  sessionColumns: SessionColumnDef[]   // Per-tool column config for Session Explorer
  dashboardSlots?: {
    overviewHero?: React.ComponentType
    rightRail?: React.ComponentType
    emptyState?: React.ComponentType
  }
  replayBlocks?: ReplayBlockRegistry
  formatSessionLabel?: (session: NormalizedSession) => string
  formatToolName?: (tool: NormalizedToolCall) => string
}

export interface AgentToolContextValue {
  toolId: AgentToolId
  definition: AgentToolDefinition
  capabilities: AgentToolCapabilities
  href: (route: string) => string
}

export interface SessionColumnDef {
  id: string
  header: string           // Uppercase HUD label
  accessor: string         // Key in data row
  sortable?: boolean
  width?: string           // CSS grid column width
}
```
